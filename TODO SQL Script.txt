1.) Create Database and Connect

2.) Create column ACCOUNT should have : firstname , lastname , email , password , tell
CREATE TABLE ACCOUNT(
EMAIL VARCHAR(30) NOT NULL PRIMARY KEY,
PASSWORD VARCHAR(20) NOT NULL,
FIRSTNAME VARCHAR(30) NOT NULL,
LASTNAME VARCHAR(30) NOT NULL,
TELL VARCHAR(10));

3.) Create column PRODUCTMEN and PRODUCTWOMEN
CREATE TABLE PRODUCTMEN(
PRODUCTCODE VARCHAR(10) NOT NULL PRIMARY KEY,
EMAIL VARCHAR(40) NOT NULL,
PRODUCTBRANDNAME VARCHAR(40) NOT NULL,
PRODUCTLINE VARCHAR(50) NOT NULL,
PRODUCTTYPE VARCHAR(50) NOT NULL,
PRODUCTSIZE INT NOT NULL,
PRODUCTPRICE DECIMAL NOT NULL,
PRODUCTDESCRIPTION VARCHAR(120),
QUANTITYINSTOCK INT , 
CONSTRAINT accountEmail_productMen FOREIGN KEY (EMAIL) REFERENCES account (EMAIL) ON DELETE NO ACTION ON UPDATE NO ACTION);



CREATE TABLE PRODUCTWOMEN(
PRODUCTCODE VARCHAR(10) NOT NULL PRIMARY KEY,
EMAIL VARCHAR(40) NOT NULL,
PRODUCTBRANDNAME VARCHAR(40) NOT NULL,
PRODUCTLINE VARCHAR(50) NOT NULL,
PRODUCTTYPE VARCHAR(50) NOT NULL,
PRODUCTSIZE INT NOT NULL,
PRODUCTPRICE DECIMAL NOT NULL,
PRODUCTDESCRIPTION VARCHAR(120),
QUANTITYINSTOCK INT , 
CONSTRAINT accountEmail_productWomen FOREIGN KEY (EMAIL) REFERENCES account (EMAIL) ON DELETE NO ACTION ON UPDATE NO ACTION);

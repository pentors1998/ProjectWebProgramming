--1.) Create Database and Connect

--2.) Create table ACCOUNT should have : firstname , lastname , email , password , tell
CREATE TABLE ACCOUNT(
EMAIL VARCHAR(30) NOT NULL PRIMARY KEY,
PASSWORD VARCHAR(20) NOT NULL,
FIRSTNAME VARCHAR(30) NOT NULL,
LASTNAME VARCHAR(30) NOT NULL,
TELL VARCHAR(10) NOT NULL,
ADDRESS VARCHAR(350) NOT NULL,
DEBITCARD VARCHAR(16) NOT NULL);

--3.) Create table ProductLine
CREATE TABLE ProductType(
PRODUCTTYPE VARCHAR(50) NOT NULL PRIMARY KEY,
TEXTDESCRIPTION VARCHAR(120));

--4.) Create table PRODUCT
CREATE TABLE PRODUCT(
PRODUCTCODE VARCHAR(10) NOT NULL PRIMARY KEY,
PRODUCTBRANDNAME VARCHAR(40) NOT NULL,
PRODUCTLINE VARCHAR(50) NOT NULL,
PRODUCTTYPE VARCHAR(50) NOT NULL,
PRODUCTSEX VARCHAR(5) NOT NULL,
PRODUCTSIZE DECIMAL NOT NULL,
PRODUCTPRICE DECIMAL NOT NULL,
PRODUCTDESCRIPTION VARCHAR(120),
QUANTITYINSTOCK INT,
CONSTRAINT Fk_productLine FOREIGN KEY (ProductType) REFERENCES ProductType (ProductType) ON DELETE NO ACTION ON UPDATE NO ACTION);

--5.) Create table HISTORYORDER
CREATE TABLE HISTORYORDER(
PRODUCTCODE VARCHAR(10) NOT NULL PRIMARY KEY,
EMAIL VARCHAR(30) NOT NULL,
PRODUCTBRANDNAME VARCHAR(40) NOT NULL,
PRODUCTLINE VARCHAR(50) NOT NULL,
PRODUCTTYPE VARCHAR(50) NOT NULL,
PRODUCTSIZE DECIMAL NOT NULL,
PRODUCTPRICE DECIMAL NOT NULL,
CONSTRAINT historyOrderProduct FOREIGN KEY (ProductCode) REFERENCES Product (ProductCode) ON DELETE NO ACTION ON UPDATE NO ACTION,
CONSTRAINT accountEmail_historyOrder FOREIGN KEY (EMAIL) REFERENCES account (EMAIL) ON DELETE NO ACTION ON UPDATE NO ACTION);
